Dimensions

Overview

Work with the dimensions of elements and browser window.

Available options:

  • width()
  • height()
  • innerWidth()
  • innerHeight()
  • outerWidth()
  • outerHeight()

Demo

Code

  • html
  • JS

<div id="div1"></div>
<p><button id="1" class="btn btn-md btn-default">Display dimensions of div</button></p>
<p><button id="2" class="btn btn-md btn-default">Display dimensions of document and window</button></p>

$(document).ready(function(){
    $("button#1").click(function(){
        var txt = "";
        txt += "Width of div: " + $("#div1").width() + "
";
        txt += "Height of div: " + $("#div1").height()+ "
";
		txt += "Outer width (+margin): " + $("#div1").outerWidth(true) + "
";
	    txt += "Outer height (+margin): " + $("#div1").outerHeight(true);
        $("#div1").html(txt);
    });
	
	$("button#2").click(function(){
        var txt = "";
        txt += "Document width/height: " + $(document).width();
        txt += "x" + $(document).height() + "\n";
        txt += "Window width/height: " + $(window).width();
        txt += "x" + $(window).height();
        alert(txt);
    });
});
